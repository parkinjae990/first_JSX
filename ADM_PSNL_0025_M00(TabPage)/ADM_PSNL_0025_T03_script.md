/**

 * @return
 */
this.fn_search = function (scholshp, jaejik)
{
		var ps = this.parent.parent.parent;

		this.ds_in.clearData();
		this.ds_in.copyData(ps.ds_in);
	
		var oData = {
		sController : ComService.$.CONTROLLER,
		svcid : "selectHjstate",
		svcList : [
			{
				action	  : ComService.$.SEARCH,
				nameSpace : "AdmPsnl0025Mapper",
				sqlId	  : "selectTab03",
				inds      : "ds_in",
				outds	  : "ds_list"
			}
		]
	};
	ComService.doAction(this, oData).then(function(form ,e){
		v_cnt = form.ds_list.rowcount;
		form.sta_cnt.set_text("총 "+ v_cnt +"건");
 	});
};
